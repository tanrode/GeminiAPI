def fibonacci(n):
  if n <= 0:
    return 0
  elif n == 1:
    return 1
  else:
    return fibonacci(n-1) + fibonacci(n-2)

def fibonacci(n):
    """
    Returns the nth fibonacci number.

    Args:
    n: The index of the fibonacci number to return.

    Returns:
    The nth fibonacci number.
    """

    # Initialize the first two fibonacci numbers.
    a = 0
    b = 1

    # Iterate over the remaining fibonacci numbers.
    for _ in range(1, n):
        # Calculate the next fibonacci number.
        c = a + b

        # Update the first and second fibonacci numbers.
        a = b
        b = c

    # Return the nth fibonacci number.
    return b
```

def binary_search(arr, target):
  """
  Performs binary search on a sorted array.

  Parameters:
    arr: The sorted array to search.
    target: The value to search for.

  Returns:
    The index of the target value in the array, or -1 if the target is not found.
  """

  left = 0
  right = len(arr) - 1

  while left <= right:
    mid = (left + right) // 2

    if arr[mid] == target:
      return mid
    elif arr[mid] < target:
      left = mid + 1
    else:
      right = mid - 1

  return -1
```
